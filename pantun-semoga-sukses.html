[tex] \color{purple} \frac{20 \: cm}{1 } = \frac{ ab }{ \frac{1}{2} \sqrt{2} } [/tex]

[tex] \color{blue} 20 \: cm \times \frac{1}{2} \sqrt{2} = bc[/tex]

[tex] \color{blue} 10\sqrt{2} \: cm = bc[/tex]

Sehingga

[tex] \color{brown} keliling \: segitiga = ab + bc + ca[/tex]
